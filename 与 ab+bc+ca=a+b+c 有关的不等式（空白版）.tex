\question

设 $ a,b,c$ 是非负实数，满足 $ab+bc+ca=a+b+c>0.$ 求证：

\begin{parts}
    \part $a+b+c+abc\geqslant 4$
    \vspace*{\stretch{1}}
    \part $a^2+b^2+c^2+5abc\geqslant 8$
    \vspace*{\stretch{1}}
    \part $a^3+b^3+c^3+13abc\geq16$
    \vspace*{\stretch{1}}
    \newpage
    \part $(a+1)(b+1)(c+1)\geqslant 8$
    \vspace*{\stretch{1}}
    \part $\sqrt{a+3}+\sqrt{b+3}+\sqrt{c+3}\geqslant 6$
    \vspace*{\stretch{1}}
    \part $\sqrt{a}+\sqrt{b}+\sqrt{c}\geqslant 2\sqrt{2}$
    \vspace*{\stretch{1}}
    \part $\dfrac{1}{a^2+b+1}+\dfrac{1}{b^2+c+1}+\dfrac{1}{c^2+a+1}\leqslant 1$
    \vspace*{\stretch{1}}
    \newpage
    \part $\sqrt{a+b+2}+\sqrt{b+c+2}+\sqrt{c+a+2}\geqslant 6$
    \vspace*{\stretch{1}}
    \part $\sqrt{ab}+\sqrt{bc}+\sqrt{ca}\geqslant 2$
    \vspace*{\stretch{1}}
    \part $\dfrac{1}{1+a^3}+ \dfrac{1}{1+b^3}+ \dfrac{1}{1+c^3}\geq\dfrac{11}{9} $
    \vspace*{\stretch{1}}
    \part $\dfrac{1}{1+a^4}+ \dfrac{1}{1+b^4}+ \dfrac{1}{1+c^4}\geq\dfrac{19}{17}$
    \vspace*{\stretch{1}}
    \newpage
    \part $a+b+c\geqslant 2+abc$
    \vspace*{\stretch{1}}
    \part $\sqrt{a+b}+\sqrt{b+c}+\sqrt{c+a}\geqslant 3\sqrt{2}$
    \vspace*{\stretch{1}}
    \part $(a+b)(b+c)(c+a)\geqslant 8$
    \vspace*{\stretch{1}}
    \part $(a^2+b)(b^2+c)(c^2+a)\geqslant 8$
    \vspace*{\stretch{1}}
    \newpage
    \part $(a^2+bc)(b^2+ca)(c^2+ab)\geqslant 8$
    \vspace*{\stretch{1}}
    \part $\dfrac{1}{3}(a+b+c)\leq(a+b-1)(b+c-1)(c+a-1)\leqslant 2(a+b+c)-5$
    \vspace*{\stretch{1}}
    \part $1<\dfrac{1}{a+b}+\dfrac{1}{b+c}+\dfrac{1}{c+a}\le\dfrac{3}{2}$
    \vspace*{\stretch{1}}
    \part $\dfrac{1}{(a+b)(a+c-1)+1}+\dfrac{1}{(b+c)(a+b-1)+1}+\dfrac{1}{(c+a)(b+c-1)+1}\le 1$
    \vspace*{\stretch{1}}
    \newpage
    \part $a+b+c+1\ge 4abc$
    \vspace*{\stretch{1}}
    \part $\dfrac{a^2}{b}+\dfrac{b^2}{c}+\dfrac{c^2}{a}+\dfrac{3n}{a^2+b^2+c^2} \geqslant 3+n,n\le 3$
    \vspace*{\stretch{1}}
    \part $\dfrac{ab+bc+ca}{3}+2\ge \sqrt{ab}+\sqrt{bc}+\sqrt{ca}$
    \vspace*{\stretch{1}}
    \part $a+b+c+1\le a^2b+b^2c+c^2a+abc$
    \vspace*{\stretch{1}}
    \newpage
    \part $\dfrac{a}{\sqrt{a+b}}+\dfrac{b}{\sqrt{b+c}}+\dfrac{c}{\sqrt{a+c}}\geqslant \dfrac{3\sqrt{2}}{2}$
    \vspace*{\stretch{1}}
    \part $\dfrac{a^2}{(b+1)(c+1)}+\dfrac{b^2}{(c+1)(a+1)}+\dfrac{c^2}{(a+1)(b+1)}\ge \dfrac{4-abc}{4}$
    \vspace*{\stretch{1}}
\end{parts}

\end{questions}
